** install visual studio code **

Go to the folder that you want to install project

** Set up virtual environment **
sudo apt-get update
sudo apt-get install python3-venv
python3 -m venv blockchain-env
source blockchain-env/bin/activate

** Check Virtual environment is properly set **
echo $VIRTUAL_ENV

** install npm**
sudo apt-get install npm

** Check npm and npx are installed properly ** 
npm -v
npx -v

**Install pip3 **
sudo apt-get install python3-pip

** Install all the dependencies required for the project **
pip3 install Flask==1.1.1
pip3 install pubnub==4.1.6
pip3 install requests==2.22.0
pip3 install cryptography==2.7
pip3 install Flask-Cors==3.0.8
pip3 install pycrypto
pip3 install pyqrcode
pip3 install pypng
pip3 install qrcode
pip3 install qrcode[pil]

pip3 install opencv-python
pip3 install pyzbar


** create frontend folder **
npx create-react-app frontend

** Create MF-QR-codes in desktop **
** copy all relevant folders and files in frontend folder **
** Go to frontend folder **
cd frontend

npm i react-bootstrap --save
npm i react-router@5.0.1 react-router-dom@5.0.1 history@4.9.0 --save
npm install react-bootstrap-table-next --save
npm install react-bootstrap-table2-filter --save

** start frontend server **
  
npm run start

** open another terminal, go to project folder, python-blockchain and start backend python server **

source blockchain-env/bin/activate
python3 -m backend.app
